// ==================== UPDATED SECTOR HEATMAP LOGIC ==================== function loadSectorHeatmap() { fetch(sectorURL + "&t=" + new Date().getTime()) .then(res => res.text()) .then(data => { // Split rows and skip the first 3 rows (Title, Empty, and Header) const rows = data.trim().split(/\r?\n/).slice(3); let rawData = []; let niftyChange = 0; rows.forEach(row => { const cols = row.split(","); // Column A is index 0 (Sector Name) // Column E is index 4 (Change %) const sector = cols[0]?.trim(); const changeStr = cols[4]?.replace("%", "").trim(); const change = parseFloat(changeStr); if (sector && !isNaN(change)) { rawData.push({ sector, change }); // Check for NIFTY 50 to use as baseline if (sector.toUpperCase() === "NIFTY 50") { niftyChange = change; } } }); // Calculate relative strength (Sector % minus Nifty %) const sectors = rawData .filter(item => item.sector.toUpperCase() !== "NIFTY 50") .map(item => ({ name: item.sector, perf: item.change - niftyChange })) .sort((a, b) => b.perf - a.perf); // Sort Descending const container = document.getElementById("sectorHeatmap"); container.innerHTML = ""; sectors.forEach(sec => { const colors = getColor(sec.perf); const tile = document.createElement("div"); tile.className = "tile"; tile.style.backgroundColor = colors.bg; tile.style.color = colors.text; tile.innerHTML = `${sec.name}${sec.perf.toFixed(2)}%`; container.appendChild(tile); }); }) .catch(err => console.error("Sector Error:", err)); }
